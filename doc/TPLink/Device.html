def reload
  res = @parent.send_data(self,
                    "system":
                     { "get_sysinfo": nil },
                    "emeter": { "get_realtime": nil })
  @rssi = res["responseData"]["system"]["get_sysinfo"]["rssi"]
  case self.class.to_s
  when 'TPLink::Light'
    reload_light(res)
  when 'TPLink::Plug'
    reload_plug(res)
  end
  true
end
